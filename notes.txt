import { useState } from 'react'

const [counter, setCounter] = useState(0)
    counter - variable
    setCounter - function
    0 - starting (number, string, method etc)

const addValue = () => {
    setCounter(counter + 1)
}

const addValue = () => {
    setCounter( (prevCounter) => prevCounter + 1)
    setCounter( (prevCounter) => prevCounter + 1)
    setCounter( (prevCounter) => prevCounter + 1)
    setCounter( (prevCounter) => prevCounter + 1)
}
